with open(datasets['train'], 'r') as f:
    data = []
    for line in f:
        parsed = json.loads(line)
        data.append({
            'published_timestamp': parsed.get('published', {}).get('$date'),
            'read_time': parsed.get('meta_tags', {}).get('twitter:data1'),
            'url': parsed.get('_id')
        })

df = (
    pd
    .DataFrame(data)
    .assign(published_timestamp = lambda df: pd.to_datetime(df['published_timestamp']))
    .assign(read_time = lambda df: df['read_time'].str.split(' ').str[0].astype(int))
    .set_index('published_timestamp')
    .sort_index()
)
